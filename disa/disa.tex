\frame {\frametitle{What is MapReduce?}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  \begin{itemize}
  \item \textbf{A programming model}:
    \begin{itemize}
    \item Inspired by functional programming
    \item Parallel computations on massive amounts of data
    \end{itemize}

\vspace{20pt}

  \item \textbf{An execution framework}:
    \begin{itemize}
    \item Designed for large-scale data processing
    \item Designed to run on clusters of commodity hardware
    \end{itemize}
 \end{itemize}
}
